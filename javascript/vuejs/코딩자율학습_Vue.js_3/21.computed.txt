<script>
export default{

  data(){
      return{
        firstName : '홍',
        lastName : '길동'
      }
  },

  computed:{
    fullName(){
      let num = 0;
      console.log('fullName 호출',num++); // 홍길동,0  (한번만 호출됨)
      return this.firstName + this.lastName + ',' + num; //홍길동,1 (한번만 호출됨)
    }
  }
}
</script>

<template> 
  <h2>{{ fullName }}</h2>  <!--로그는 처음 한번만 출력된다, 의미는 한번만 호출한다는 것.-->
  <h2>{{ fullName }}</h2>
  <h2>{{ fullName }}</h2>
</template>
