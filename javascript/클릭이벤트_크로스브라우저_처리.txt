function foo(e) {


   // 이벤트 객체 얻기
   var event = e ? e : window.event;

   //링크 이동 막기
   if (event.stopPropagation) {
      // 넷스케이프 계열의 경우
      event.stopPropagation();
      event.preventDefault();
   } else {
      // IE의 경우
      event.cancelBubble = true;
   }

   // 이벤트 타겟 얻기, 클릭객체, this
   var target = event.target ? event.target : event.srcElement;

   console.log($(target).text()); // 여기

}
