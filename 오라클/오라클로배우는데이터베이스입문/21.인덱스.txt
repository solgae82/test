인덱스 검색 방식 구분
	Table Full Scan 방식과 Index Scan으로 구분 한다

인덱스도 오라클 데이터베이스 객체이므로 소유 사용자와 사용 권한이 존재한다

인덱스는 사용자가 직접 특정 테이블의 열에 지정할 수도 있지만, 열이 기본키,또는 unique key일때 
자동으로 생성된다.

인덱스 조회 (현재 계정 소유)
	-- 현재 계정 소유 인덱스 정보 
	select * from USER_INDEXES;

	-- 현재 계정 소유 인덱스 컬럼 정보
	select * from USER_IND_COLUMNS;

인덱스 생성
	형식 : CREATE INDEX '인덱스명' ON EMP('컬럼명' [ASC/DESC]);

	-- EMP 테이블 SAL 컬럼에 'IDX_EMP_SAL' 인덱스명 생성
	CREATE INDEX IDX_EMP_SAL ON EMP(SAL);

	-- EMP 테이블 SAL 컬럼에 'IDX_EMP_SAL' 내림차순 인덱스명 생성
	CREATE INDEX IDX_EMP_SAL ON EMP(SAL DESC);

인덱스 삭제	
	DROP INDEX IDX_EMP_SAL;

인덱스 종류
	단일 인덱스(single index)
		: 1개 열로 이뤄진 인덱스 

		create index idx_name 
		on emp(sal); 

	복합 인덱스(concatenated index),결합 인덱스(composite index)
		: 두개 이상 열로 만들어지는 인덱스

		create index idx_name 
		on emp(sql,ename);

	고유 인덱스(unique index)
		: 열에 중복 데이터가 없을때 사용

		create unique index idx_name 
		on emp(empno);

	함수 기반 인덱스(function based index)
		: 열에 산술식 등 데이터 가종이 진행된 결과로 인덱스 생성
		
		create index idx_name 
		on emp(sal * 12 + comm);

	비트맵 인덱스(bitmap index)
		: 데이터 종류가 적고 같은 데이터가 많이 존재할 때 주로 사용

		create bitmap index idx_name 
		on emp(job);
