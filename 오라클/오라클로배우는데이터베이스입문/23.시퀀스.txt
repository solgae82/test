sys,system 접속으로 계정에 시퀀스 생성 권한을 줘야한다
	GRANT CREATE SEQUENCE TO DONG;

-- 테스트 테이블 생성
	create table dept_sequence 
	as select * from dept where 1 <> 1;

	select * from dept_sequence;

-- 시퀀스 생성
	create sequence SEQ_DEPT_SEQUENCE 
	INCREMENT BY 10
	START WITH 10
	MAXVALUE 90
	MINVALUE 0
	NOCYCLE
	CACHE 2;

-- 생성한 시퀀스 조회
	select * from user_sequences;

-- dept_sequence 테이블에 시퀀스(.nextval)로 데이터 넣기
	insert into dept_sequence (deptno , dname, loc) 
	values(seq_dept_sequence.nextval, 'DATABASE', 'SEOUL');

-- dept_sequence 테이블 조회
	select * from dept_sequence;

-- 현재 seq_dept_sequence 시퀀스(.currval) 값 조회
	select seq_dept_sequence.currval from dual;

-- 시퀀스 수정
	ALTER SEQUENCE SEQ_DEPT_SEQUENCE 
	INCREMENT BY 3
	MAXVALUE 99
	CYCLE;

-- 시퀀스 삭제
	DROP SEQUENCE SEQ_DEPT_SEQUENCE;

형식 :
	create sequence '시퀀스이름'
	[increment by '숫자'] 
		: 시퀀스 증감 값(기본값 1)
	
	[start with '숫자'] 
		: 시퀀스 시작 값(기본값 1)	
	
	[maxvalue '숫자' | nomaxvalue] 
		: 시퀀스 최대값 지정,nomaxvalue 일때(오름차순=10*27승,내림차순=-1) 
	
	[minvalue '숫자' | nominvalue] 
		: 시퀀스 최소값 지정,nominvalue 일때(오름차순=1,내림차순=10 *(-26)승)
	
	[cycle : nocyle] 
		: cycle은 최대값 도달후 다시 시작값으로 돌아가서 증감, nocycle은 중단 후 에러발생
	
	[cache '숫자' | nocache] 
		: 시퀀스 증감값 cache에 할당해 놓는 수 지정,nocache는 증감값 캐시 중단, 
		설정하지 않으면 디폴트 20.

	[order | noorder] 
		: RAC 환경시에는 order, 단일 환경시에는 noorder


--------- order 옵션 

오라클 시퀀스 ORDER 옵션은 여러 데이터베이스 노드에서 동시에 시퀀스 값을 생성해야 할 때, 
생성된 번호의 순서를 보장하는지 여부를 설정합니다. 

ORDER 옵션을 사용하면 각 노드에서 생성되는 번호가 순차적으로 보장되어 순서가 뒤섞이지 않지만, 
캐싱이 비활성화되거나 성능이 저하될 수 있습니다. 

반면, NOORDER 옵션은 각 노드에서 독립적으로 번호를 생성하여 성능을 최적화하며, 
순서가 보장되지 않는 단점이 있습니다. 

ORDER 옵션
	ORDER: 시퀀스 값의 순서가 보장됩니다. RAC(Real Application Clusters) 환경에서 
	각 노드가 시퀀스 값을 생성할 때 번호가 뒤섞이지 않도록 합니다.
	
	성능: 번호 순서를 보장하기 위해 ORDER 옵션을 사용하면 시퀀스 캐시가 비활성화되어 성능이 저하될 수 있습니다. 

NOORDER 옵션
	NOORDER: 시퀀스 값의 순서가 보장되지 않습니다. 각 노드에서 독립적으로 번호를 생성하여 성능을 최적화합니다.
	성능: 기본값이며, 성능을 우선시하는 환경에서 사용됩니다. 순서 보장이 필요 없는 경우에 적합합니다.
	가장 일반적: 대부분의 시나리오에서 NOORDER이 기본값으로 사용되며, 성능에 더 유리합니다. 
선택 가이드
	순서 보장 필요 시: RAC 환경에서 번호의 순서가 반드시 중요한 경우 ORDER 옵션을 선택합니다.
	성능 우선 시: 순서 보장보다 성능이 더 중요한 경우 NOORDER 옵션을 선택합니다.
