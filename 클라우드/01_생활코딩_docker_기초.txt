Docker 컨테이너 기술은 리눅스 운영체제 기술이다.
	

기본 구조 
	docker hub : 공식 Registry (https://hub.docker.com/)
	image : docker image를 통해 컨테이너가 구동된다.
		(실행에 필요한 프로그램, 라이브러리,미들웨어,OS,네트워크 설정등을 패키징한 image)
	container : 실행되는 컨테이너	

docker hub(registry) -> (pull) -> image -> (run) -> container

---------

# httpd image pull 하기

	docker pull httpd	

# pull한 로컬 image들 모두 보기

	docker images	

# httpd 이미지 컨테이너로 등록 후 시작하기
	#docker run은 내부적으로 pull + create + start 과정을 모두 포함합니다
	
	docker run httpd 

# httpd 이미지를 'ws1' 이라는 이름을 설정 후 run하기
	
	docker run --name ws1 httpd
	
# 컨테이너 프로세서 목록 보기
	
	docker ps	#현재 실행중인 컨테이너 목록 출력
	docker ps -a	#정지+실행중인 컨테이너 목록 출력  
	
# 컨테이너 등록된 ws1 정지 및 시작하기
	
	docker stop ws1
	docker start ws1
	
# 컨테이너 ws1 로그 보기
	
	docker logs -f ws1
	
# 컨테이너 삭제하기(컨테이너 정지 후 삭제가능)

	docker stop ws1 
	docker rm ws1
	
# 컨테이너 삭제하기(정지 + 삭제 동시에 하기)

	docker rm --force ws1
	
# 이미지 삭제하기 (해당 이미지로 등록된 컨테이너 모두 삭제 후 삭제 가능)

	docker rmi httpd

# host port 와 컨테이너 port 연결 설정하기
	# run 할때 설정한다
	-p 옵션 : (host 포트번호):(container 포트번호)
	
	docker run --name ws1 -p 8080:80 httpd
	
# 컨테이너 내부 shell로 명령하기(컨테이너 실행 후 가능)

	docker exec -it ws1 /bin/sh  # or /bin/bash
	
	# pwd # 컨테이너 내부에서 /bin/sh 쉘로 내린 pwd 명령
	# exit # 컨테이너 내부에서 /bin/sh 쉘로 종료 명려 (빠져나오기) 

# volume 설정 (컨테이너와 호스트 파일시스템 연결)
	#run 할때 설정한다
	#호스트에서 앱 소스를 변경하면 컨테이너 내부의 애플리케이션도 변경되는 설정이다
	
	-v 옵션 : (host 디렉토리 볼륨):(container 내부 디릭토리 볼륨)
	
	docker run --name ws1 -p 8080:80 -v d:/docker:/usr/local/apache2/htdocs httpd 
	
# 커스텀한 컨테이너를 image로 만들기
	#이미 커스텀된 'ws1' 을 이미지로 만들기
	
	$ docker commit ws1 ws1-server

	$ docker images #생성된 이미지 확인
	
		..
		ws1-server:latest   4a2f3aececa6        175MB         45.2MB
