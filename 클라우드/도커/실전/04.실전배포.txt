1 PC 에서 이미지 빌드
	
	$ cd D:\flask-app
	dir
	
	flask-app/
	├── Dockerfile
	├── docker-compose.yml
	├── requirements.txt
	└── server.py
	
	server.py
		from flask import Flask

		app = Flask(__name__)

		@app.route('/')
		def hello():
		    return "Hello, Docker and Flask!"

		if __name__ == '__main__': # 이거 중요함
		    # 로컬 테스트용 (Docker로 띄울 때는 Gunicorn이 대신 실행합니다)
		    app.run(host='0.0.0.0', port=5000)

	Dockerfile
		# 1. 파이썬 3.12 슬림 버전으로 시작
		FROM python:3.12-slim

		# 2. 컨테이너 내부 작업 폴더 설정
		WORKDIR /app

		# 3. 패키지 목록을 먼저 복사하고 설치 (빌드 캐시 최적화)
		COPY requirements.txt .
		RUN pip install --no-cache-dir -r requirements.txt

		# 4. 나머지 내 플라스크 소스코드 전부 복사
		COPY . .

		# 5. 컨테이너가 켜지면 gunicorn 웹서버로 플라스크 실행 (포트 5000)
		# server.py 를 실행하는데, app변수가 gunicorn이 다룰 변수명임을 알려줌.
		CMD ["gunicorn", "--bind", "0.0.0.0:5000", "server:app"]

	docker-compose.yml
		# version: '3.8' # 최신 버전엔 필요 없어졌다고 함

		services:
		  flask-web:
		    build: .                 # 현재 폴더(.)에 있는 Dockerfile을 읽어서 빌드해라
		    image: solgae/my-flask-app:1.0  # 이미지명:버전
		    container_name: flask_app
		    ports:
		      - "5000:5000"          # 내 컴퓨터 포트(앞) : 컨테이너 내부 포트(뒤)
		    volumes:
		      - .:/app               # [꿀팁] 내 컴퓨터의 코드 수정이 컨테이너에 실시간 반영됨
		    environment:
		      - FLASK_ENV=development


	$ docker compose up # 도커 이미지 만들기
	$ docker images # 이미지 생성 확인
		..
		solgae/my-flask-app:1.0  
	

2. 도커 허브에 올리기 
	# 도커 허브 로그인
	$ docker login
	
	$ docker push solgae/my-flask-app:1.0
	
	# 올린 후 도커 허브 사이트에 가서 업로드 잘 되었는지 확인
	
3. 리눅스 서버 
	$ cd /home/solgae
	
	# 실행용 컴포즈 파일 생성 
	# 이미지 생성할때 옵션을 덮어쓰기 하면서 컨테이너 생성
	
	vi docker-compose.yml
	services:
	  flask-web:
	    image: solgae/my-flask-app:1.0
	    container_name: flask_app
	    ports:
	      - "5000:5000"
	    environment:
	      - FLASK_ENV=production

	:wq

	# 항상 docker-compose.yml 파일 있는 곳에 와서 실행해야 함.

	docker compose up # 최조 처음 실행

	#운영
	docker compose stop	# 정지
	docker compose start	# 시작
	docker compose restart  # 재시작

-------------------
예전에 되었지만, 지금은 안되는 명령어
❌ docker-compose ... (2023년 공식 종료된 구버전 V1 명령어)
⭕ docker compose ... (띄어쓰기! 현재 표준이자 앞으로 계속 쓸 최신 V2 명령어)

gunicorn 사용시 여러개 워커 프로세서로 인해 발생하는 (나오다 안 나오다가)문제가 있을 수 있다
# --workers=1 옵션을 추가합니다.(성능 저하를 감수 해야할 코드)
CMD ["gunicorn", "--workers=1", "--bind", "0.0.0.0:5000", "server:app"]
