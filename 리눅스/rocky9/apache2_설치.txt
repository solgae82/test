1. 사전 도구 설치

	sudo yum groupinstall "Development Tools"
	sudo yum -y install gcc gcc-c++ make expat-devel

	#로키10은 pcre2 를 설치해야 한다 (pcre -> pcre2)
	sudo yum -y install pcre2-devel

2. apr , apr-util , pcre2 , apache2.4 다운로드
	
	/usr/local/src 에 다운로드 
		apr-1.7.6.tar.gz
		apr-util-1.6.3.tar.gz
		pcre2-main.zip # 로키10에서는 pcre2 를 써야한다고 함.
		httpd-2.4.66.tar.gz
	
3. apr 설치

	cd /usr/local/src
	tar xvfz apr-1.7.6.tar.gz
	cd apr-1.7.6
	./configure --prefix=/usr/local/apr
	
	make
	make install
	cd /usr/local
	ls # apr 디렉토리 생겼는지 확인 & 디렉토리도 들어가서 설치파일 잘 생성되었는지 확인	
	
4. apr-util 설치
	
	cd /usr/local/src
	tar xvfz apr-util-1.6.3.tar.gz
	cd apr-1.6.3
	./configure --prefix=/usr/local/apr-util --with-apr=/usr/local/apr
	
	make
	make install
	cd /usr/local
	ls # apr-util 디렉토리 생겼는지 확인 & 디렉토리도 들어가서 설치파일 잘 생성되었는지 확인	
	
5. pcre2 설치 

	cd /usr/local/src
	unzip pcre2-main.zip
		
	cd pcre2-main
	./autogen.sh # 실행해서 configure 파일 생성
	./configure --prefix=/usr/local/pcre2
	
	make
	make install
	cd /usr/local
	ls # pcre2 디렉토리 생겼는지 확인 & 디렉토리도 들어가서 설치파일 잘 생성되었는지 확인	

6. openssl 위치 확인
	
	whereis openssl
	openssl: /usr/include/openssl
		
7. apach 설치

	cd /usr/local/src
	tar xvfz httpd-2.4.66.tar.gz
	cd httpd-2.4.66

	./configure --prefix=/usr/local/apache2.4 \
	--enable-mods-shared=all \
	--enable-module=so \
	--enable-rewrite \
	--enable-so \
	--enable-ssl \
	--with-ssl=/usr/include/openssl \
	--with-apr=/usr/local/apr \
	--with-apr-util=/usr/local/apr-util \
	--with-pcre2=/usr/local/pcre2/bin/pcre2-config

	make
	make install
	cd /usr/local
	ls # apache2.4 디렉토리 생겼는지 확인 & 디렉토리도 들어가서 설치파일 잘 생성되었는지 확인
	
	
	
===================
아파치 소스 컴파일 시 SSL(--enable-ssl) 설정을 빼먹었다면 기본적으로 재컴파일하는 것이 가장 확실하고 권장되는 방법입니다.
그러나 현재 위 설치는 
	mod_ssl 모듈만 추가 (DSO 방식) 가능한 방법으로 컴파일 되었으므로
	apxs 로 모듈 빌드해서 추가 가능하다 (검색해서 적용하라)
