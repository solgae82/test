오라클 함수는 내장함수 와 사용자 정의함수 로 나뉜다.

내장함수는 단일행함수, 다중행 함수로 세분화된다.

집계함수는 다중행 함수이다

-- sum 
	필드의 합계 반환
	sum([distinct, all, 설정 않음(선택, 기본 all)] , 열이나 데이터)
	NULL값은 계산하지 않는다

	select sum(sal)  from emp;  // 29025 
	select sum(sal) ,ename from emp; //에러 , ename처럼 여러열 출력 필드를 같이 쓸수 없음.

	select sum(distinct sal) ,sum(all sal) , sum(sal) from emp; // 24775(중복제거)	29025	29025 

-- count
	데이터 갯수 반환
	count([distinct, all, 설정 않음(선택, 기본 all)] , 열이나 데이터)
	NULL값은 계산하지 않는다

	select count(*) from emp; // 14
	select count(sal) from emp;	// 14

	select count(distinct sal), 
	    count(all sal),
	    count(sal)
	from emp;
	// 12(중복제거)	14	14


	select count(comm) from emp; // 4(null은 카운트하지 않는다)
	select count(*) from emp where comm is not null; // 4

-- MAX , MIN
	최대값, 최솟값 반환
	MAX, MIN([distinct, all, 설정 않음(선택, 기본 all)] , 열이나 데이터)
	옵션을 설정하나 안하나 값은 같기에 설정할 필요가 없다.

	select MAX(sal), MIN(sal) from emp; // 5000, 800
	select MAX(hiredate),MIN(hiredate)  from emp; // 87/05/23	80/12/17

-- AVG
	평균값 구하기
	NULL은 계산에 포함하지 않는다(평균 카운트에도 제외된다)

	select comm from emp; //14개 행과 4개의  값외에 나머지는  NULL

	//avg (comm) = sum(comm) / 4
	select avg(comm), avg(all comm) from emp; // 550 550

	//총 6개행에 중복 1 제거 = 5열
	select sum(distinct sal), count(distinct sal) , avg(distinct sal) from emp where deptno=30;
	// 8150		5	1630  ( 8150 / 5 = 1630)
