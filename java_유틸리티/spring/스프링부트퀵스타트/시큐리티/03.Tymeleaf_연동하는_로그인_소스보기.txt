------------------- SecurityUser.java (시큐리티 관련, 커스텀 UserDetails 구현 확장)

package com.solgae.boardweb.security;

import java.util.Collection;
import java.util.List;

import org.jspecify.annotations.Nullable;
import org.springframework.security.core.GrantedAuthority;
import org.springframework.security.core.authority.SimpleGrantedAuthority;
import org.springframework.security.core.userdetails.UserDetails;

import com.solgae.boardweb.domain.Member;

public class SecurityUser implements UserDetails {

	private static final long serialVersionUID = 987489524972825590L;
	
	private Member member;
	
	
	public SecurityUser(Member member){
		this.member = member;
	}
	
	@Override
	public Collection<? extends GrantedAuthority> getAuthorities() {
		
		return List.of(new SimpleGrantedAuthority(member.getRole().toString()));
	}

	@Override
	public @Nullable String getPassword() {
		return member.getPassword();
	}

	@Override
	public String getUsername() {
		return member.getId();
	}

	public Member getMember() {
		return member;
	}

	public void setMember(Member member) {
		this.member = member;
	}
	
}

------------------- SecurityUserDetailsService.java  (시큐리티 관련, 커스텀 UserDetailsService 구현)

package com.solgae.boardweb.security;

import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.security.core.userdetails.User;
import org.springframework.security.core.userdetails.User.UserBuilder;
import org.springframework.security.core.userdetails.UserDetails;
import org.springframework.security.core.userdetails.UserDetailsService;
import org.springframework.security.core.userdetails.UsernameNotFoundException;
import org.springframework.stereotype.Service;

import com.solgae.boardweb.domain.Member;
import com.solgae.boardweb.persistence.MemberRepository;

/**
 * security 가 로그인 여부를 판단하는 service
 *
 */
@Service
public class SecurityUserDetailsService implements UserDetailsService {

	@Autowired
	MemberRepository memberRepo;
	
	@Override
	public UserDetails loadUserByUsername(String username) throws UsernameNotFoundException {
		
		Optional<Member> findById = memberRepo.findById(username);
		
		if(!findById.isPresent()) {
			throw new UsernameNotFoundException(username);
		}
	
		Member member = findById.get();
		
		return new SecurityUser(member);
	}

}

------------------- SecurityConfig.java (시큐리티 Config 설정)

package com.solgae.boardweb.security;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.security.web.SecurityFilterChain;

@Configuration
public class SecurityConfig {

	@Bean
	SecurityFilterChain filterChain(HttpSecurity secutiry) throws Exception{
		
		secutiry.authorizeHttpRequests(authorize->authorize
				.requestMatchers("/","/system/**").permitAll() //모두가 접근할 수 있는 path
				.requestMatchers("/board/**").authenticated() //인증된 사용자만 접근가능
				.requestMatchers("/admin/**").hasRole("ADMIN") //인증 + 'ADMIN' role일때만 접근가능한 path
				.anyRequest().authenticated() //나머지 path는 인증되었을때만 접근가능
			);

			
		//403 Forbidden 페이지 설정
		secutiry.exceptionHandling(exception->exception
					.accessDeniedPage("/system/accessDenied")); 
			
		// CSRF 비활성화 , 실전에서 이렇게 하면 안된다
		secutiry.csrf(csrf -> csrf.disable()); 
			
		//로그인 설정
		secutiry.formLogin(form-> form.loginPage("/system/login")
				.defaultSuccessUrl("/board/getBoardList",true));
		
		//로그아웃 설정
		secutiry.logout(form->form.logoutUrl("/system/logout")
				.invalidateHttpSession(true)
				.logoutSuccessUrl("/"));
		
			
		return secutiry.build();
	}
	
	// 비밀번호 암호화 Bean 등록 (시큐리티가 이 암호방식을 가져다 쓴다)
	@Bean
	public PasswordEncoder passwordEncoder() {
		return new BCryptPasswordEncoder();
	}
}
