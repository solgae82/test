스프링 데이터 JPA(Spring Data JPA)는 스프링 프레임워크에서 
JPA(Java Persistence API) 기반 데이터 접근 계층 개발을 단순화하는 기술로, 
개발자가 반복적인 CRUD(Create, Read, Update, Delete) 코드를 줄이고 비즈니스 로직에 집중할 수 있게 도와줍니다. 

핵심은 JpaRepository 인터페이스 제공을 통해 데이터베이스 접근을 객체 지향적으로 처리하며, 
SQL 작성 부담을 크게 줄여준다는 점입니다.

------------------- Board.java

package com.solgae.domain;

import java.time.LocalDateTime;

import jakarta.persistence.Entity;
import jakarta.persistence.GeneratedValue;
import jakarta.persistence.Id;
import lombok.Getter;
import lombok.Setter;
import lombok.ToString;

@Getter
@Setter
@ToString
@Entity
public class Board {

	@Id
	@GeneratedValue
	private Long seq;
	private String title;
	private String writer;
	private String content;
	private LocalDateTime createDate;
	private Long cnt;
}
